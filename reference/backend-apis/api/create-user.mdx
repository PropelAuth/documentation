---
title: "Create user"
api: "POST https://{auth_url}/api/backend/v1/user/"
---

### Path Parameters

<ParamField path="auth_url" type="string" required>
  The domain for your application
</ParamField>

### JSON Body Parameters

<ParamField body="email">The user's email address</ParamField>

<ParamField body="email_confirmed">
  Whether the email address should start off as already confirmed. If false, the
  user is required to confirm the email address before they sign in.
</ParamField>

<ParamField body="send_email_to_confirm_email_address">
  Whether we should send an email immediately to confirm the user's email
  address. If false, the user will get a confirmation email when they first sign
  in.
</ParamField>

<ParamField body="password">
  Optional password. If unset, the user can set it themselves via their account
  page
</ParamField>

<ParamField body="username">
  Optional username. Can only be used if username is enabled in your user schema
</ParamField>

<ParamField body="first_name">
  Optional first name. Can only be used if name is enabled in your user schema
</ParamField>

<ParamField body="last_name" >
Optional last name. Can only be used if name is enabled in your user schema

</ParamField>

{
  "user_id": "b5f667fb-e51a-49c6-a396-711e62948689"

</ResponseExample>
